161static void measureAndShowTask(void *pvParameter){
162 while (true)
163 {
164 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
165
166 uint16_t distancia = 0;
167 if (MEDIR == true) {
168 //distancia = 1 + rand() % 39; // Simulate a distance between 1 and 40 cm
169 distancia = HcSr04ReadDistanceInCentimeters();
170 #if DEBUG_MODE
171 printf("Distancia: %u cm\n", distancia); // Print the measured distance
172 // printf("Estado... \n");
173 // printf("\tMEDIR: %d\n", MEDIR); // Print the state of MEDIR
174 // printf("\tHOLD: %d\n", HOLD); // Print the state of HOLD
175 #endif
176 }
177 if (HOLD == false && MEDIR == true) {
178 encenderLedsSegunDistancia(distancia);
179 LcdItsE0803Write(distancia);
180 #if DEBUG_MODE
181 printf("LCD: %u cm\n", distancia);
182 #endif
183 }
184 }
185}
