232static void measureAndShowTask(void *pvParameter){
233 while (true)
234 {
235 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
236
237 if (MEDIR == true) {
238 distancia = HcSr04ReadDistanceInCentimeters();
239 // distancia = 1 + rand() % 39; // Simulate a distance between 1 and 40 cm
240 // distancia = 16; // Simulate a specific distance
241 #if DEBUG_MODE && SHOW_PRINTFS_ON
242 printf("Distancia: %lu cm\n", distancia); // Print the measured distance
243 #endif
244
245 if (HOLD == false) {
246 if ((SHOW_MAX && (distancia > distance_to_send)) || !SHOW_MAX) {
247 distance_to_send = distancia;
248 }
249 encenderLedsSegunDistancia(distance_to_send);
250 LcdItsE0803Write(distance_to_send);
251 #if DEBUG_MODE && SHOW_PRINTFS_ON
252 printf("LCD: %lu cm\n", distancia);
253 #endif
254 }
255 UartSendDistance(distance_to_send); // Send the distance via UART
256 }
257 }

293void uartListen(void *pvParameter) {
294 uint8_t data;
295 UartReadByte(UART_PC, &data);
296 switch (data) {
297 case 'O':
298 if_TEC1_do();
299 break;
300 case 'H':
301 if_TEC2_do();
302 break;
303 case 'I':
304 if (UNIT == CM)
305 { UNIT = INCHES; }
306 else
307 { UNIT = CM; }
308 break;
309 case 'M':
310 SHOW_MAX = !SHOW_MAX;
311 #if DEBUG_MODE && TOGGLE_PRINTFS_ON
312 printf("\nToggled MAX from: %d --> %d\n", !SHOW_MAX, SHOW_MAX);
313 #endif
314 break;
315 case 'F':
316 if (timer_measure_config.period > DELAY_MEASURE_MIN) {
317 timer_measure_config.period -= DELAY_MEASURE_STEP;
318 TimerUpdatePeriod(timer_measure_config.timer, timer_measure_config.period);
319 }
320 #if DEBUG_MODE && TOGGLE_PRINTFS_ON
321 printf("\nToggled DELAY from: %lu ms --> %lu ms\n", (timer_measure_config.period + DELAY_MEASURE_STEP)/1000, timer_measure_config.period /1000);
322 #endif
323 break;
324 case 'S':
325 if (timer_measure_config.period < DELAY_MEASURE_MAX) {
326 timer_measure_config.period += DELAY_MEASURE_STEP;
327 TimerUpdatePeriod(timer_measure_config.timer, timer_measure_config.period);
328 }
329 #if DEBUG_MODE && TOGGLE_PRINTFS_ON
330 printf("\nToggled DELAY from: %lu ms --> %lu ms\n", (timer_measure_config.period - DELAY_MEASURE_STEP)/1000, timer_measure_config.period /1000);
331 #endif
332 break;
333 default:
334 break;
335 }
336}
