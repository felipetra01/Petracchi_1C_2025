124void encenderLedsSegunDistancia(uint16_t distancia) {

156static void measureAndShowTask(void *pvParameter){
157 while (true)
158 {
159 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
160
161 uint16_t distancia = 0;
162 if (MEDIR == true) {
163 //distancia = 1 + rand() % 39; // Simulate a distance between 1 and 40 cm
164 distancia = HcSr04ReadDistanceInCentimeters();
165 #if DEBUG_MODE
166 printf("Distancia: %u cm\n", distancia); // Print the measured distance
167 #endif
168 }
169 if (HOLD == false && MEDIR == true) {
170 encenderLedsSegunDistancia(distancia);
171 LcdItsE0803Write(distancia);
172 #if DEBUG_MODE
173 printf("LCD: %u cm\n", distancia);
174 #endif
175 }
176 }
177}
