162static void medirMostrarTask(void *pvParameter){
163 while (true)
164 {
165 uint16_t distancia = 0;
166 if (MEDIR == true) {
167 //distancia = 1 + rand() % 39; // Simulate a distance between 1 and 40 cm
168 distancia = HcSr04ReadDistanceInCentimeters();
169 #if DEBUG_MODE
170 printf("Distancia: %u cm\n", distancia); // Print the measured distance
171 printf("Estado... \n");
172 printf("\tMEDIR: %d\n", MEDIR); // Print the state of MEDIR
173 printf("\tHOLD: %d\n", HOLD); // Print the state of HOLD
174 printf("\tSwitches: %d\n",SwitchesRead());
175 #endif
176 }
177 if (HOLD == false) {
178 encenderLedsSegunDistancia(distancia);
179 LcdItsE0803Write(distancia);
180 #if DEBUG_MODE
181 printf("LCD: %u cm\n", distancia);
182 #endif
183 }
184 #if DEBUG_MODE
185 printf("Fin medicion\n"); // Print end of measurement message
186 #endif
187
188 vTaskDelay(DELAY_MEDICION / portTICK_PERIOD_MS); // Delay for 1 second
189
190 #if DEBUG_MODE
191 printf("Fin delay de 1 segundo\n"); // Print delay message
192 #endif
193 }
194}

200static void teclasTask(void *pvParameter){
201 while (true) {
202 if (SwitchesRead() == TEC1) {
203 MEDIR = !MEDIR;
204 if (MEDIR == false) {
205 LcdItsE0803Off(); // Apagar el LCD
206 encenderLedsSegunDistancia(1); // Apagar todos los LEDs
207 }
208 }
209 if (SwitchesRead() == TEC2 && MEDIR == 1) {
210 HOLD = !HOLD;
211 }
212 vTaskDelay(DELAY_TASK / portTICK_PERIOD_MS); // Delay for 100 ms
213 }
214}
